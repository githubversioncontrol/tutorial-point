Ant - Jacoco Integration - this is basic - update this with full example

Steps
  • Add jacocoant.jar to classpath
  • Update build.xml
Steps

<?xml version="1.0" encoding="UTF-8"  xmlns:jacoco="antlib:org.jacoco.ant"?>

<target name="jacocoTestReport" depends="junitTest">
	<mkdir dir="junit/coverage-report"/>
	
	<taskdef name="junit" classname="org.apache.tools.ant.taskdefs.optional.junit.JUnitTask">
		<classpath path="path/jacocoant.jar" />
	</taskdef>
	
	<jacoco:coverage destfile="junit/jacoco.exec">
		<junit fork="yes" printsummary="withOutAndErr">
			<classpath location="WEB-INF/classes"/>
			<formatter type="xml" />
			<test name="com.test.MyTest" todir="junit"/>
			<classpath refid="classpath.compile"/>
		</junit>
	</jacoco:coverage>
	
	<jacoco:report>
		<executiondata>
			<fil file="junit/jacoco.exec"/>
		</executiondata>
		<structure name="jacoco-sample">
			<classfiles>
				<fileset dir="WEB-INF/classes"/>
			</classfiles>
			<sourcefiles encoding="UTF-8">
				<fileset dir="src" />
			</sourcefiles>
		</structure>
		<html destdir="junit/coverage-report"/>
	</jacoco:report>
</target>
