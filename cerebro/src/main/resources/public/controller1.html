<!DOCTYPE html>
<html ng-app="moduleName">
<head>
<meta charset="ISO-8859-1">
<title>Controller 1</title>

<script type="text/javascript" src="controller1.js"></script>
</head>
<body ng-controller="mainController">
<div>
First Name: { { jsObjectDetails.str1 } }
</div>
<div>
Last Name: { {jsObjectDetails.str2} }
</div>
<div>
<img ng-src="{ {jsObjectDetails.imgSrc} }" title="{ {jsObjectDetails.str1} } { {jsObjectDetails.str2} }"/>

controller1.js

(function () {

    var app = angular.module("moduleName", []);

    app.controller("mainController", function ($scope) {
        var jsObject = {
            str1: "Cerebro",
            str2: "Angular Controller1",
            imgSrc: "img1.jpg" // path to jpg file
        }

        $scope.jsObjectDetails = jsObject;
    });

}());
