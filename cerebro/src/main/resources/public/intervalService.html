intervalService.html

<!DOCTYPE html>
<html ng-app="moduleName">

<head>
    <meta charset="ISO-8859-1">
    <title>interval service</title>
    <script type="text/javascript" src="https://ajax.googleapis.com/ajax/libs/angularjs/1.5.6/angular.min.js"></script>
    <script type="text/javascript" src="intervalService.js"></script>
</head>

<body ng-controller="mainController">
    <h3 style="text-align: center;">{ {message} }</h3>
    <hr/> Count Down : { {countDown} }
    <hr/>
    <h4>{ {intervalCancelMessage} }</h4>
    <input type="button" value="Cancel Interval" ng-click="cancelInterval()" />
</body>

</html>
intervalService.js

(function () {
    var app = angular.module("moduleName", []);

    app.controller("mainController", function ($scope, $http, $interval) {
        $scope.message = "Welcome to Angular interval service demo";
        $scope.countDown = 10;
        $scope.intervalCancelMessage = "";

        $scope.countDownTimer = function () {
            $scope.countDown -= 1;
        }

        // 3rd parameter is how many times interval has to be executed. In below example interval will be disabled after 10 times execution
        var countDownInterval = $interval($scope.countDownTimer, 1000, 10);


        $scope.cancelInterval = function () {
            $interval.cancel(countDownInterval);
            $scope.intervalCancelMessage = "Interval cancelled, refresh the page to start interval again";
        }

    })
}());
