<mapper namespace="Student">

	<!-- Read queries -->
	<resultMap id="allColumnsResultMap" type="StudentAlias">
		<result property="id" column="ID"/>
		<result property="name" column="NAME"/>
		<result property="course" column="COURSE"/>
		<result property="courseStartDate" column="START_DATE"/>
		<result property="courseEndDate" column="END_DATE"/>
	</resultMap>
	
	<resultMap id="nameAndCourseColumns" type="Student2Alias">
		<result property="name" column="NAME"/>
		<result property="course" column="COURSE"/>
	</resultMap>
	
	<!-- get all records -->
	<select id="getAll" resultMap="allColumnsResultMap">
		select * from student
	</select>
	
	<select id="getById" parameterType="int" resultMap="allColumnsResultMap">
		select * from student where id = #{id}
	</select>
	
	<select id="getByNameLike" parameterType="string" resultMap="allColumnsResultMap">
		select * from student where name like #{inputName}
	</select>
	
	<select id="getByNameAndCourse" parameterType="map" resultMap="allColumnsResultMap">
		select * from student where name like #{inputName} and course like #{inputCourse}
	</select>
	
	<select id="getNameAndCourse" resultMap="nameAndCourseColumns">
		select name, course from student
	</select>
	
</mapper>

Student.java

package com.mybatis.crud.practice;

import java.util.Date;

public class Student {

	private Integer id;
	private String name;
	private String course;
	private Date courseStartDate;
	private Date courseEndDate;

	public Student() {
	}

	public Student(Integer id, String name, String course, Date courseStartDate, Date courseEndDate) {
		this.id = id;
		this.name = name;
		this.course = course;
		this.courseStartDate = courseStartDate;
		this.courseEndDate = courseEndDate;
	}

	public Integer getId() {
		return id;
	}

	public void setId(Integer id) {
		this.id = id;
	}

	public String getName() {
		return name;
	}

	public void setName(String name) {
		this.name = name;
	}

	public String getCourse() {
		return course;
	}

	public void setCourse(String course) {
		this.course = course;
	}

	public Date getCourseStartDate() {
		return courseStartDate;
	}

	public void setCourseStartDate(Date courseStartDate) {
		this.courseStartDate = courseStartDate;
	}

	public Date getCourseEndDate() {
		return courseEndDate;
	}

	public void setCourseEndDate(Date courseEndDate) {
		this.courseEndDate = courseEndDate;
	}

	@Override
	public String toString() {
		return "Student (id=" + id + ", name=" + name + ", course=" + course + ", courseStartDate="
				+ courseStartDate + ", courseEndDate=" + courseEndDate + ")";
	}
}

package com.mybatis.crud.practice;

import java.io.IOException;
import java.io.Reader;
import java.util.Date;

import org.apache.ibatis.io.Resources;
import org.apache.ibatis.session.SqlSession;
import org.apache.ibatis.session.SqlSessionFactory;
import org.apache.ibatis.session.SqlSessionFactoryBuilder;
import org.junit.Test;

public class InsertPractice {

	@Test
	public void insert() throws IOException {
		Reader reader = Resources.getResourceAsReader("MyBatisConfiguration.xml");
		SqlSessionFactory sqlSessionFactory = new SqlSessionFactoryBuilder().build(reader);

		try (SqlSession sqlSession = sqlSessionFactory.openSession();) {
			Student student = new Student(21, "Jack", "Java", new Date(), new Date());

			// insert record
			sqlSession.insert("Student.insertQuery", student);
			sqlSession.commit();
		}
	}
}

public void update1() throws IOException {
		Reader reader = Resources.getResourceAsReader("MyBatisConfiguration.xml");
		SqlSessionFactory sqlSessionFactory = new SqlSessionFactoryBuilder().build(reader);
		try (SqlSession sqlSession = sqlSessionFactory.openSession()) {
			Map<String, String> inputMap = new HashMap<>();
			inputMap.put("inputCourse", "MyBatis2");
			inputMap.put("inputName", "Stott");

			// read records
			List<Student> list = sqlSession.selectList("Student.getByNameLike", "Stott");
			System.out.println("----- Before -------");
			list.stream().forEach(System.out::println);

			list = sqlSession.selectList("Student.getByNameLike", "Stott");
