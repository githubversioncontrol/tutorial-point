App.js
import React, { Component } from 'react';
import './App.css';

class App extends Component {

  constructor(){
    super();
    this.state = {
      todos: [],
      title: "React Todo Application",
      counter: 0
    }
  }

  addTodo = (event) => {
    event.preventDefault();
    let name = this.refs.name.value;
    let completed = this.refs.completed.value;
    let counter = this.state.counter;
    counter = counter + 1;

    let todo = {
      name,
      completed,
      counter
    }

    let todos = this.state.todos;
    todos.push(todo);
    this.setState({
      todos: todos,
      counter: counter
    });

    this.refs.todoForm.reset();
  }

  removeTodo = (index) => {
    let todos = this.state.todos;
   
    let newTodos = [];
    for (let i = 0; i < todos.length; i++) {
      let val1 = todos[i];
      if (val1.counter !== index) {
        newTodos.push(val1);
      }
    }

    this.setState({
      todos: newTodos
    });
  }

  render() {
    let todos = this.state.todos;
    return (
      <div className="App">
        <h3>{this.state.title}</h3>
        <hr />
        <form ref="todoForm">
          <input type="text" placeholder="Enter Text" ref="name" />
          <input type="text" placeholder="Status" ref="completed" />
          <button type="button" onClick={this.addTodo}>Add Todo </button>
        </form>
        <hr />
        <ul>
          {todos.map(todo => <li key={todo.counter}>{todo.name}-{todo.completed}
              <button onClick={this.removeTodo.bind(null, todo.counter)}>Remove Todo</button></li>)}
        </ul>
      </div>
    );
  }
}

export default App;

1. If you are using visual code - View - Integrated Terminal - npm start
2. Application will be started on port 3000
3. Open browser http://localhost:3000
