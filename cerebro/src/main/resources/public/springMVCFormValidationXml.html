MyController.java

package mvc.form.validation.controller;

import javax.validation.Valid;

import org.springframework.stereotype.Controller;
import org.springframework.validation.BindingResult;
import org.springframework.web.bind.annotation.ModelAttribute;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestMethod;
import org.springframework.web.servlet.ModelAndView;

import mvc.form.validation.model.Employee;

@Controller
public class MyController {

	@RequestMapping(value = "/showForm", method = RequestMethod.GET)
	public ModelAndView showForm(ModelAndView modelAndView) {
		modelAndView.setViewName("myForm");
		modelAndView.addObject("employeeModel", new Employee());
		return modelAndView;
	}

	@RequestMapping(value = "/submit", method = RequestMethod.POST)
	public ModelAndView submitForm(@Valid @ModelAttribute("employeeModel") Employee employee,
			BindingResult bindingResult) {

		System.out.println("has errors: " + bindingResult.hasErrors());

		ModelAndView modelAndView = new ModelAndView();
		if (bindingResult.hasErrors()) {
			modelAndView.setViewName("myForm");
		} else {
			modelAndView.setViewName("empDetails");
			modelAndView.addObject("employee", employee);
		}

		return modelAndView;
	}
}
Employee.java

package mvc.form.validation.model;

import javax.validation.constraints.NotNull;

import org.hibernate.validator.constraints.NotEmpty;
import org.hibernate.validator.constraints.Range;

public class Employee {

	@NotEmpty
	private String firstName;
	@NotEmpty
	private String lastName;

	@NotNull
	@Range(min = 21, max = 60)
	private Integer age;

	public Employee() {
	}

	public Employee(String firstName, String lastName, Integer age) {
		this.firstName = firstName;
		this.lastName = lastName;
		this.age = age;
	}

	public String getFirstName() {
		return firstName;
	}

	public void setFirstName(String firstName) {
		this.firstName = firstName;
	}

	public String getLastName() {
		return lastName;
	}

	public void setLastName(String lastName) {
		this.lastName = lastName;
	}

	public Integer getAge() {
		return age;
	}

	public void setAge(Integer age) {
		this.age = age;
	}

	@Override
	public String toString() {
		return "Employee [firstName=" + firstName + ", lastName=" + lastName + ", age=" + age + "]";
	}

}
