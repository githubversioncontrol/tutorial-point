import React from 'react';

class App extends React.Component{
    constructor(){
        super();
        this.state = {
            data:
            [
                {
                    "id": 1,
                    "name": "jack",
                    "course": "Java"
                },
                {
                    "id": 2,
                    "name": "jill",
                    "course": "Angular"
                },
                {
                    "id": 3,
                    "name": "john",
                    "course": "React"
                }
            ]
        }
    }
    render(){
        let tableStyle = {
            border: '1px solid black'
        };

        return (
            <div>
                <hr />
                <h3> Header Component</h3>
                <Header />
                
                <hr />
                <h3>Content Component </h3>
                <Content />

                <hr />
                <h3> Display Table </h3>
                <table style={tableStyle}>
                    <tbody>
                        {this.state.data.map((person, i) => <TableRow key={i} data={person}/>)}
                    </tbody>
                </table>

                <hr />
            </div>
        );
    }
}

class Header extends React.Component{
    render(){
        return(
            <div>
                <h2>Header </h2>
            </div>
        );
    }
}

class Content extends React.Component{
    render(){
        let i = 10;

        let myStyle = {
            color: '#F00'
        };

        return(
            <div> 
                <h1>Hello World</h1>
                <h2>Hello World 2</h2>
                <h3>Hello World 3</h3>
                <p style={myStyle}>This is paragraph with style attribute</p>    
                {10+10}

                <h3>{i > 5 ? 'TRUE' : 'FALSE'}</h3>
                <h3>{i > 10 ? 'TRUE' : 'FALSE'}</h3>
            </div>
        );
    }
}

class TableRow extends React.Component{
    render(){
        let tableStyle = {
            border: '1px solid black'
        };

        return(
            <tr style={tableStyle}>
             <td style={tableStyle}>{this.props.data.id}</td>
             <td style={tableStyle}>{this.props.data.name}</td>
             <td style={tableStyle}>{this.props.data.course}</td>
            </tr>
        );
    }
}

export default App;
