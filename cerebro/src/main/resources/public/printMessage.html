Steps
  • Create printMessage.html file
  • Add angular.js file in script tag
  • Create printMessage.js file
  • Add printMessage.js to printMessage.html in script tag
  • Create message variable. Add it to $scope in printMessage.js
  • Print message in printMessage.html
printMessage.html

<!DOCTYPE html>
<html ng-app="moduleName">
<head>
<meta charset="ISO-8859-1">
<title>Print Message</title>
<script type="text/javascript" src="https://ajax.googleapis.com/ajax/libs/angularjs/1.4.8/angular.min.js"></script>
<script type="text/javascript" src="printMessage.js"></script>
</head>
<body ng-controller="mainController">
<h3>{{message} }</h3>
</body>
</html>
printMessage.js

// moduleName is name of the module
// [] dependencies for this module
var app = angular.module('moduleName', []);

app.controller('mainController', function($scope){
    $scope.message = "Angular JS Hello World";
});
