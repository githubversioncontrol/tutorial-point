template.js

(function(){
    var app = angular.module("moduleName", ["ngRoute"]);

    app.config(function($routeProvider){
        $routeProvider
                .when("/default", {templateUrl: "default.html", controller: "mainController"})
                .when("/page1", {templateUrl: "page1.html", controller: "mainController"})
                .when("/page2", {templateUrl: "page2.html", controller: "mainController"})
                .when("/page3/:param1", {templateUrl: "page3.html", controller: "mainController"})
                .otherwise({redirectTo: "/default"});
    });

    app.controller("mainController", function($scope, $routeParams){
        $scope.message = "Welcome to Angular Template and Routing";
        $scope.param1 = $routeParams.param1;
    });

}());
