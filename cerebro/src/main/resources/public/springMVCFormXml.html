MyController.java

package mvc.form.controller;

import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.ModelAttribute;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestMethod;
import org.springframework.web.servlet.ModelAndView;

import mvc.form.model.Employee;

@Controller
public class MyController {

	@RequestMapping(value = "/form", method = RequestMethod.GET)
	public ModelAndView form1(@ModelAttribute("employeeModel") Employee employee, ModelAndView modelAndView) {
		modelAndView.setViewName("form1");
		return modelAndView;
	}

	@RequestMapping(value = "/addEmployee", method = RequestMethod.POST)
	public ModelAndView addEmployee(@ModelAttribute("employeeModel") Employee employee, ModelAndView modelAndView) {
		System.out.println("employee: " + employee);
		modelAndView.setViewName("results");
		modelAndView.addObject("employee", employee);
		return modelAndView;
	}
}
Employee.java

package mvc.form.model;

public class Employee {

	private String firstName;
	private String lastName;
	private String designation;

	public Employee(String firstName, String lastName, String designation) {
		this.firstName = firstName;
		this.lastName = lastName;
		this.designation = designation;
	}

	public Employee() {
	}

	public String getFirstName() {
		return firstName;
	}

	public void setFirstName(String firstName) {
		this.firstName = firstName;
	}

	public String getLastName() {
		return lastName;
	}

	public void setLastName(String lastName) {
		this.lastName = lastName;
	}

	public String getDesignation() {
		return designation;
	}

	public void setDesignation(String designation) {
		this.designation = designation;
	}

	@Override
	public String toString() {
		return "Employee [firstName=" + firstName + ", lastName=" + lastName + ", designation=" + designation + "]";
	}

}
