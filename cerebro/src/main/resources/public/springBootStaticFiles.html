Using Static contents like html, CSS, javascript and images

Steps
  • Create a core java project using maven
  • Create folder named public under src/main/resources
  • Create folders named CSS (contains all css files), imgs (contains all images), js (contains all js files)
  • Create index.html under public folder and refer above created files from css, imgs, js folders
Maven command

mvn archetype:generate -DgroupId=spring.boot.static.content -DartifactId=SpringBootStaticContent -Dversion=1.0 -DarchetypeArtifactId=maven-archetype-quickstart -DinteractiveMode=false

index.html

<!DOCTYPE html>
<html>
<head>
<meta charset="ISO-8859-1">
<title>Spring Boot HTML</title>
<link rel="stylesheet" type="text/css" href="css/index.css">
<script type="text/javascript" src="js/index.js"></script>
</head>
<body>
	Welcome to Spring Boot using Static HTML<br/><br/>
	
	<div>Using div from CSS</div><br/>
	
	<button value="Popup" onclick="buttonClicked()">Popup</button><br/><br/>
	
	<img alt="tom-jerry" src="imgs/tom-jerry.jpg">
</body>
</html>
StaticContentController.java

package com.spring.boot.controler;

import org.springframework.http.MediaType;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestMethod;
import org.springframework.web.bind.annotation.RestController;

@RestController
public class StaticContentController {

	@RequestMapping(value="/method1", method=RequestMethod.GET, produces=MediaType.TEXT_PLAIN_VALUE)
	public String method1(){
		return "StaticController method1()";
	}
}
StaticContentApp.java

package com.spring.staticContent;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.context.annotation.ComponentScan;

@SpringBootApplication
@ComponentScan(basePackages={"com.spring"})
public class StaticContentApp {

	public static void main(String[] args) {
		SpringApplication.run(StaticContentApp.class, args);
	}
}
