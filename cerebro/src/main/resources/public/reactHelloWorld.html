webpack.config.js
Explanation 
Open webpack.config.js file and add the following code. We are setting webpack entry point to be main.js. 
Output path is the place where bundled app will be served. We are also setting the development server to 8080 port. 
You can choose any port you want.

And lastly, we are setting babel loaders to search for js files, and use es2015 and react presets that 
we installed before.

var config = {
   entry: './main.js',
   output: {
      path:'/',
      filename: 'index.js',
   },
   devServer: {
      inline: true,
      port: 9092
   },
   module: {
      loaders: [
         {
            test: /\.jsx?$/,
            exclude: /node_modules/,
            loader: 'babel-loader',
            query: {
               presets: ['es2015', 'react']
            }
         }
      ]
   }
}
module.exports = config;

App.jsx
import React from 'react';

class Button extends React.Component {
    render() {
        return (
            <button type="button">{this.props.name}</button>
        );
    }
}

class StudentList extends React.Component {
    render() {
        return (
            <div>
                <table>
                    <thead>
                        <tr>
                            <th>Firstname</th>
                            <th>Lastname</th>
                            <th>Age</th>
                        </tr>
                    </thead>
                    <tbody>
                        <tr>
                            <td>Jill</td>
                            <td>Smith</td>
                            <td>19</td>
                        </tr>
                        <tr>
                            <td>Eve</td>
                            <td>Jackson</td>
                            <td>21</td>
                        </tr>
                    </tbody>
                </table>
            </div>
        );
    }
}

class App extends React.Component {
    render() {
        let i = 10;

        let myStyle = {
            color: '#F00'
        };

        return (
            <div>
                <h1>Hello World</h1>
                <h2>Hello World 2</h2>
                <h3>Hello World 3</h3>
                <p style={myStyle}>This is paragraph with style attribute</p>    
                {10+10}

                <h3>{i > 5 ? 'TRUE' : 'FALSE'}</h3>
                <h3>{i > 10 ? 'TRUE' : 'FALSE'}</h3>

                <hr />
                <h3>Buttons:</h3>
                <Button name="Add" />
                <Button name="Edit" />
                <Button name="Delete" />

                <hr />
                <h1>Student List</h1>
                <StudentList />
            </div>
        );
    }
}

export default App;
