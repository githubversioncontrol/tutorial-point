Steps
  • Create ngModel.html file
  • Add angular.js in html script tag
  • Create ngModel.js and add it to ngModel.html script tag
ngModel.html

<!DOCTYPE html>
<html ng-app="moduleName">

<head>
    <meta charset="ISO-8859-1">
    <title>Ng Model</title>
    <script type="text/javascript" src="https://ajax.googleapis.com/ajax/libs/angularjs/1.5.6/angular.min.js"></script>
    <script type="text/javascript" src="ngModel.js"></script>
</head>

<body ng-controller="mainController">
    <h3 style="text-align:center;">{{message}}</h3>

    <hr/>
    <h3>Without Initialized text box:</h3>
    <h4>{ {userInput} }</h4>
    <form name="ngModelDemo">
        <input type="text" placeholder="Enter any text" ng-model="userInput" />
    </form>

    <hr/>
    <h3>Pre Initialize value to be displayed in text box: </h3>
    <h4>{ {userInput2} }</h4>
    <form name="ngModel2">
        <input type="text" placeholder="Enter any text" ng-model="userInput2"/>
    </form>
</body>

</html>
ngModel.js

(function () {
    var app = angular.module("moduleName", []);

    app.controller("mainController", function ($scope) {

        $scope.message = "Welcome to Angular ngModel";

        $scope.userInput2 = "User Input 2"
    });
}());
