(function(){
    var app = angular.module("moduleName", []);

    app.controller("mainController", function($scope, $location, $anchorScroll, $log){
        $scope.message = "Welcome to Angular location and anchorScroll demo to navigate to specific id in a page";

        $scope.navigateToId = function(id){
            $log.info(id);
            $location.hash(id);
            $anchorScroll();
        }
    });
}());
