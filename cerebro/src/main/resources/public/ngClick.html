Steps
  • Create ngClick.html
  • Create ngClick.js. Add it in ngClick.html
ngClick.html

<!DOCTYPE html>
<html ng-app="moduleName">

<head>
    <meta charset="ISO-8859-1">
    <title>ngClick</title>
    <script type="text/javascript" src="https://ajax.googleapis.com/ajax/libs/angularjs/1.5.6/angular.min.js"></script>
    <script type="text/javascript" src="ngClick.js"></script>
</head>

<body ng-controller="mainController">
    <h3 style="text-align:center;">{ {message} }</h3>
    <hr/>
    <h3> {{buttonClickMessage}} </h3>
    <input type="button" value="Click Me" ng-click="buttonClicked()" />
</body>

</html>
ngClick.js

(function () {
    var app = angular.module("moduleName", []);

    app.controller("mainController", function ($scope) {
        $scope.message = "Welcome to Angular ngClick";

        $scope.buttonClicked = function () {
            $scope.buttonClickMessage = "Button Clicked";
        };
    });
}());
