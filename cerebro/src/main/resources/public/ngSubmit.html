Steps
  • Create ngSubmit.html
  • Create ngSubmit.js. Add it in ngSubmit.html
ngSubmit.html

<!DOCTYPE html>
<html ng-app="moduleName">

<head>
    <meta charset="ISO-8859-1">
    <title>ngSubmit</title>
    <script type="text/javascript" src="https://ajax.googleapis.com/ajax/libs/angularjs/1.5.6/angular.min.js"></script>
    <script type="text/javascript" src="ngSubmit.js"></script>
</head>

<body ng-controller="mainController">
    <h3 style="text-align:center;">{ {message} }</h3>
    <hr/>
    <h3>{ {formSubmittedMessage} }</h3>
    <form name="ngSubmit Form" ng-submit="formSubmitted()">
        <input type="text" placeholder="Enter mandatory text field" ng-model="enteredText" required/>
        <input type="submit" value="Submit" />
    </form>
</body>

</html>
ngSubmit.js

(function(){
    var app = angular.module("moduleName",[]);

    app.controller("mainController", function($scope){
        $scope.message = "Welcome to Angular ngSubmit Directive Demo";

        $scope.formSubmitted = function(){
            $scope.formSubmittedMessage = "Form Submitted with text: "+$scope.enteredText;
        }
    });

}());
