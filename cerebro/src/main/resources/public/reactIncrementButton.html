// Button component to display button
class Button extends React.Component{
	
  // to handle button click
  handleClick = () => {
  	this.props.onClickIncrementCounter(this.props.amountIncrementValue);
  }
  
  render(){
  	return(
      	<button onClick={this.handleClick}> 
        	+{this.props.amountIncrementValue} 
        </button>
    );
  }

}

// component to display result
class Result extends React.Component{
	render(){
  	return(
    	<div>Counter Value: {this.props.counterValue}</div>
    );
  }
}

// app component - container for Button, Result components
class App extends React.Component{
	state = {
  	counter: 0
  }
  
  incrementCounter = (incrementValue) => {
  	this.setState((prevState) => ({
    	counter: prevState.counter + incrementValue
    }));
  }
  
	render(){
  	return(
    	<div>
      	<Button amountIncrementValue={1} onClickIncrementCounter={this.incrementCounter}/>
        <Button amountIncrementValue={5} onClickIncrementCounter={this.incrementCounter}/>
        <Button amountIncrementValue={10} onClickIncrementCounter={this.incrementCounter}/>
        <Button amountIncrementValue={100} onClickIncrementCounter={this.incrementCounter}/>
        <Result counterValue={this.state.counter}/>
      </div>
    );
  }
}

ReactDOM.render(<App />, mountNode);
