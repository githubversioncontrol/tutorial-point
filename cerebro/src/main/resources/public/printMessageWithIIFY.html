Steps
  • Create printMessageWithIIFY.html file
  • Add angular.js file in script src
  • Create printMessageWithIIFY.js file
  • Add printMessageWithIIFY.js to html in script tag
  • Create message variable. Add it to $scope in printMessageWithIIFY.js
  • Print message in printMessageWithIIFY.html using angular directive
printMessage.html

<!DOCTYPE html>
<html ng-app="moduleName">
<head>
<meta charset="ISO-8859-1">
<title>IIFY</title>
<script type="text/javascript" src="https://ajax.googleapis.com/ajax/libs/angularjs/1.5.6/angular.min.js"></script>
<script type="text/javascript" src="printMessageWithIIFY.js"></script>
</head>
<body ng-controller="controllerName">
We are using Angular 1.5.6 for this example.<br/>
Angular 1.4.x is giving an error if we use IIFY
<h3>{{message} }</h3>
</body>
</html>
printMessage.js

(function () {
    var app = angular.module("moduleName", []);

    app.controller("controllerName", function ($scope) {
        $scope.message = "Welcome to IIFY. IIFY: Immediately Invoked Functional Expression. IIFY is used to avoid global variables";
    });
}());
