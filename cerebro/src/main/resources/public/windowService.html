Steps
  • Create windowService.html
  • Create windowService.js. Add it in windowService.html
windowService.html

<!DOCTYPE html>
<html ng-app="moduleName">
<head>
<meta charset="ISO-8859-1">
<title>window service</title>
<script type="text/javascript" src="https://ajax.googleapis.com/ajax/libs/angularjs/1.5.6/angular.min.js"></script>
<script type="text/javascript" src="windowService.js"></script>
</head>
<body ng-controller="mainController">
<h3 style="text-align: center;"> { {message} } </h3>
<hr/>
Count Down: { {countDown} }
</body>
</html>
windowService.js

(function(){
    var app = angular.module("moduleName", []);

    app.controller("mainController", function($scope, $window, $http, $interval){
        $scope.message = "Welcome to Angular window service demo";
        $scope.countDown = 5;

        $scope.countDownTimer = function(){
            $scope.countDown -= 1;
            if($scope.countDown == 0){
                $window.location.reload();
            }
        }

        $interval($scope.countDownTimer, 1000, 5);
    });
}());
