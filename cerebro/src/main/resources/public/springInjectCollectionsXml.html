package spring.core.hello.world.controller;

import spring.core.hello.world.service.MyService;

public class MyController {

MyRepository.java

package spring.core.hello.world.repository;

import java.util.List;
import java.util.Map;
import java.util.Properties;
import java.util.Set;

public class MyRepository {

	private Map<String, String> myMap;
	private List<String> myList;
	private Set<String> mySet;
	private Properties properties;

	public Map<String, String> getMyMap() {
		return myMap;
	}

	public void setMyMap(Map<String, String> myMap) {
		this.myMap = myMap;
	}

	public List<String> getMyList() {
		return myList;
	}

	public void setMyList(List<String> myList) {
		this.myList = myList;
	}

	public Set<String> getMySet() {
		return mySet;
	}

	public void setMySet(Set<String> mySet) {
		this.mySet = mySet;
	}

	public Properties getProperties() {
		return properties;
	}

	public void setProperties(Properties properties) {
		this.properties = properties;
	}

	public void method1() {
		System.out.println("MyRepository -> method1(): ");

		// iterate Map
		System.out.println("iterate myMap");
		myMap.forEach((key, value) -> System.out.println(key + value));

		// iterate list
		System.out.println("iterate myList");
		myList.stream().forEach(element -> System.out.println(element));

		// iterate set
		System.out.println("iterate mySet");
		mySet.stream().forEach(element -> System.out.println(element));

		// iterate properties
		System.out.println("iterate properties");
		Set<Map.Entry<Object, Object>> props = properties.entrySet();
		props.stream().forEach(entry -> System.out.println(entry.getKey() + ", " + entry.getValue()));
	}
}
MyApp.java

package spring.core.hello.world;

import org.springframework.context.support.AbstractApplicationContext;
import org.springframework.context.support.ClassPathXmlApplicationContext;

import spring.core.hello.world.controller.MyController;

public class MyApp {
	public static void main(String[] args) {
		AbstractApplicationContext applicationContext = new ClassPathXmlApplicationContext("springConfig.xml");

		MyController myController = (MyController) applicationContext.getBean("myController");
		myController.method1();

		applicationContext.close();
	}
}
